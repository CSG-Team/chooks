# useMount

仅在组件挂载时执行一次


## 使用
```tsx
import { useState } from 'react';
import { useMount } from 'cchooks';

const MyComponent = () => {
  useMount(() => {
  message.info('mount');
  });

  return <div>Hello csg</div>;
};

const Mount = () => {
  const [boolean, setBoolean] = useState(false);

  return (
    <>
      <button type="button" onClick={() => setBoolean(!boolean)}>
        {boolean ? 'unmount' : 'mount'}
      </button>
      {boolean && <MyComponent />}
    </>
  );
};

```
